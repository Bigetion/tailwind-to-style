## 🚀 How to Use

```bash
npm install tailwind-to-style@3.2.0
```

```javascript
import {
  // Core
  tws,
  twsx,
  
  // Presets
  applyPreset,
  materialPreset,
  
  // Plugins
  usePlugin,
  gradientPlugin,
  
  // Optimizer
  optimizeClasses,
  
  // Animations
  createAnimation,
  animations,
  
  // Composition
  button,
  card,
  box,
  
  // SSR
  createStyleCollector,
  
  // Validation
  validateWithWarnings,
  autoFix,
  
  // DevTools
  enableDevTools,
  createDebugPanel,
} from 'tailwind-to-style';

// Apply Material Design
applyPreset(materialPreset);

// Use gradient plugin
usePlugin(gradientPlugin);

// Enable DevTools
enableDevTools({ logPerformance: true });
createDebugPanel();

// Optimize classes
const classes = optimizeClasses('flex p-4 p-6 bg-primary-500');

// Create animation
const fadeIn = animations.fadeIn('300ms');

// Use composition API
const btnProps = button({ variant: 'solid', colorScheme: 'primary' });

// Validate and fix
const fixed = autoFix('flex-center text-middle');

// Generate styles
const styles = tws(classes);
```
